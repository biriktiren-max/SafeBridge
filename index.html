CÜZDAN HATTINI BAĞLAMA FONKSİYONU async function connectWallet() { if (typeof window.ethereum !== 'undefined') { try { // MetaMask'ı doğrudan hesapları getirmeye zorluyoruz, askıda kalmayı önler const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' }); const walletAddress = accounts[0]; // Ekrandaki yeşil butonu güncelle const connectBtn = document.getElementById('connectButton'); // Buton ID'nize göre ayarlayın if (connectBtn) { connectBtn.innerText = `HAT BAĞLANDI: ${walletAddress.substring(0, 6).toUpperCase()}...`; connectBtn.style.backgroundColor = '#28a745'; } logMesaji("Cüzdan hattı başarıyla bağlandı abi."); } catch (error) { logMesaji("Cüzdan bağlantısı reddedildi veya hata oluştu."); console.error(error); } } else { logMesaji("MetaMask bulunamadı, lütfen tarayıcı eklentisini kontrol et abi."); } } // 2. BLOCKCHAIN KİLİDİNİ VURMA (İŞLEMİ TETİKLEME) FONKSİYONU async function createTransaction() { logMesaji("İşlem başlatılıyor, veriler kontrol ediliyor..."); try { // Girdileri alıyoruz const aliciInput = document.getElementById('aliciAdresi').value.trim(); let miktarInput = document.getElementById('miktarInput').value.trim(); const sifreInput = document.getElementById('guvenlikSifresi').value; // BÜYÜK ARIZA ÇÖZÜMÜ: Virgülü otomatik olarak noktaya çeviren emniyet kilidi miktarInput = miktarInput.replace(',', '.'); const miktarFloat = parseFloat(miktarInput); // Girdi Kontrolleri if (!aliciInput || isNaN(miktarFloat) || miktarFloat <= 0 || !sifreInput) { logMesaji("İŞLEM İPTAL: Lütfen tüm alanları eksiksiz ve doğru doldur abi."); return; } // MetaMask'ın bağlı olduğu güncel hesabı tekrar çekip emin oluyoruz const accounts = await window.ethereum.request({ method: 'eth_accounts' }); if (accounts.length === 0) { logMesaji("İŞLEM İPTAL: Cüzdan hattı kopuk, önce cüzdanı bağla abi."); return; } const gonderenAdresi = accounts[0]; // DÖNGÜ HATASI KONTROLÜ: Gönderen ve alıcı aynı olamaz if (gonderenAdresi.toLowerCase() === aliciInput.toLowerCase()) { logMesaji("İŞLEM İPTAL: Alıcı adresi olarak kendi adresini yazamazsın abi."); return; } // Blockchain Akıllı Sözleşme Çağrısı (Web3 veya Ethers.js yapınıza göre) // miktarFloat değerini web3.utils.toWei(miktarInput, 'ether') şeklinde kullanabilirsin abi. logMesaji("MetaMask onay penceresi tetikleniyor, lütfen sağ paneli kontrol et abi..."); // Bu aşamada akıllı sözleşme fonksiyonunuz (Örn: contract.methods.createTransaction(...).send({...})) çağrılacak. } catch (error) { console.error(error); logMesaji("İŞLEM İPTAL: Bakiye yetersiz veya ağda sorun var abi."); } } // Sitedeki log kutusuna yazdırma yardımcısı function logMesaji(mesaj) { const logKutusu = document.getElementById('logKutusu'); // Log panelinizin ID'si if (logKutusu) { const zaman = new Date().toLocaleTimeString('tr-TR'); logKutusu.innerHTML += `
[${zaman}] > ${mesaj}`; } }
